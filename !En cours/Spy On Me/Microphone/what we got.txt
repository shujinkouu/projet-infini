using sounddevice package in python

import sounddevice as sd

# this gets the devices of your computer
>>> sd.query_devices()

#this selects frequency
>>> fs = 44100

#setting up duration of the recording
duration = 10.5 # seconds
#this records and saves it to an array "myrecording"
>>> myrecording = sd.rec(int(duration * fs), samplerate=fs, channels=2)
#this wait until the recording is done it is also a way to check if this is recording
>>> sd.wait()
#this plays the array stored in the variable "myrecording" here it is our recording
>>> sd.play(myrecording, fs)

so far we can record audio and play it back to the user, can't save it locally yet
